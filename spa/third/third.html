            angular.module('SpaApp')

            .config(function($routeProvider) {
                $routeProvider.
                when('/first', {
                    templateUrl: 'first/first.html',
                    controller: 'FirstCtrl'
                }).
                when('/second', {
                    templateUrl: 'second/second.html',
                    controller: 'SecondCtrl'
                }).
                when('/third/:id?', {
                    templateUrl: 'third/third.html',
                    controller: 'ThirdCtrl'
                }).
                otherwise({
                    redirectTo: '/first'
                });
            })


            .controller('ThirdCtrl', function($scope, $routeParams) {
                $scope.paramId = $routeParams.id || 'no param';
                $scope.content = 'ThirdCtrl scope content variable';
            });

        
 
            <p>
                <label>Parameter: </label>
                <span>{{paramId}}</span>
            </p>
